并行计算

  • 多线程
  • GPU computing - PyCuda
  • Message Passing Interface (MPI)

from mpi import MPI
mpi = MPI()

def pow(x):
    return x**2

mpi.begin()

x = range(10)

result = mpi.parfor(pow, x)

mpi.end()
